#include <stdio.h>

void multiplyPolynomials(int poly1[], int m, int poly2[], int n, int result[]) {
    for (int i = 0; i < m + n - 1; i++) {
        result[i] = 0;
    }
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            result[i + j] += poly1[i] * poly2[j];
        }
    }
}

void printPolynomial(int poly[], int n) {
    for (int i = n - 1; i >= 0; i--) {
        printf("%dx^%d", poly[i], i);
        if (i > 0) {
            printf(" + ");
        }
    }
    printf("\n");
}

int main() {
    int m, n;

    printf("Enter the degree of Polynomial 1: ");
    scanf("%d", &m);

    int poly1[m + 1];
    printf("Enter the coefficients of Polynomial 1 (from x^0 to x^%d): ", m);
    for (int i = 0; i <= m; i++) {
        scanf("%d", &poly1[i]);
    }

    printf("Enter the degree of Polynomial 2: ");
    scanf("%d", &n);

    int poly2[n + 1];
    printf("Enter the coefficients of Polynomial 2 (from x^0 to x^%d): ", n);
    for (int i = 0; i <= n; i++) {
        scanf("%d", &poly2[i]);
    }

    int result[m + n];

    multiplyPolynomials(poly1, m + 1, poly2, n + 1, result);

    printf("Polynomial 1: ");
    printPolynomial(poly1, m + 1);

    printf("Polynomial 2: ");
    printPolynomial(poly2, n + 1);

    printf("Resultant Polynomial: ");
    printPolynomial(result, m + n);

    return 0;
}
